# 5. Сконфигурируйте ansible на сервере BR-SRV:
------------------------------------------------------------------------------------
### 5.1 Сформируйте файл инвентаря, в инвентарь должны входить HQ-SRV, HQ-CLI, HQ-RTR и BR-RTR
### 5.2 Рабочий каталог ansible должен располагаться в /etc/ansible
### 5.3 Все указанные машины должны без предупреждений и ошибок отвечать pong на команду ping в ansible посланную с BR-SRV.


*Настройка br-srv*
------------------------------------------------------------------------------------ 
```bash

apt-get install -y ansible sshpass


mcedit /etc/ansible/ansible.cfg в нем убираем комментарии
inventory = /etc/ansible/hosts
host_key_checking = false


mcedit /etc/ansible/hosts

(sh ip int br) (ip a)
HQ-SRV ansible_host=192.168.0.10 ansible_user=sshuser ansible_password=P@ssw0rd ansible_port=2026
HQ-CLI ansible_host=192.168.1.3 ansible_user=sshuser ansible_password=P@ssw0rd ansible_port=2026
HQ-RTR ansible_host=10.10.10.1 или 172.16.0.1 ansible_user=admin ansible_password=admin ansible_connection=network_cli ansible_network_os=ios
BR-RTR ansible_host=192.168.3.1 ansible_user=admin ansible_password=admin ansible_connection=network_cli ansible_network_os=ios
[all:vars]
ansible_python_interpreter=/usr/bin/python3


ansible-galaxy collection install ansible.netcommon
ansible-galaxy collection install cisco.ios
```


*Настройка br-rtr и hq-rtr*

```bash
en
conf t
security none
end

wr mem
```

*Настройка HQ-CLI*

```bash
useradd -u 2026 -G wheel sshuser
echo Port 2026 >> /etc/openssh/sshd_config
echo AllowUsers sshuser >> /etc/openssh/sshd_config
passwd sshuser
systemctl restart network
systemctl restart sshd
```


*Проверка*

```bash
ansible -m ping all
```




*Роксана*
```bash

mcedit /etc/ansible/ansible.cfg в нем убираем комментарии
inventory = /etc/ansible/inventory.yml
host_key_checking = false
```
*В файле ssd_config добавляем порты для пользователя на hq-srv*

*В файле inventory.yml*
```bash
HQ-RTR ansible_host=192.168.3.1 ansible_user=admin ansible_password=admin ansible_connection=network_cli ansible_network_os=ios
BR-RTR ansible_host=192.168.4.1 ansible_user=admin ansible_password=admin ansible_connection=network_cli ansible_network_os=ios
HQ-SRV ansible_host=192.168.3.2 ansible_user=term_access ansible_password=P@ssw0rd ansible_port=2203
HQ-CLI ansible_host=192.168.3.66 ansible_user=user1 ansible_password=P@ssw0rd ansible_port=2203

[all:vars]
ansible_python_interpreter=/usr/bin/python3
```

*в командной строке BR-SRV*
```bash
ansible-galaxy collection install cisco.ios
apt-get install python3-module-pip
pip3 install ansible-pylibssh
```
